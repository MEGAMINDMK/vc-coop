SetEntityPos


Syntax

SetEntityPos()

Example

local ped = AddPed(10,20,10.5)
local pedX,pedY,pedZ = GetEntityPos(ped)
print("PedX: "..pedX.." PedY: "..pedY.." PedZ: "..pedZ)

local veh = AddVehicle(226, 10, 20, 10.5)
local vehX,vehY,vehZ = GetEntityPos(veh)
print("vehX: "..vehX.." vehY: "..vehY.." vehZ: "..vehZ)
